BinOp = enum Mul | Div | Add | Sub
Expr = type value: Int
     | infix: (op: BinOp * Expr * Expr)

eval: Expr -> Int
eval v: Expr.value = v
eval e: Expr.infix = match infix.op
                     to Mul => eval infix.1 * eval infix.2
                     to Div => eval infix.1 / eval infix.2
                     to Add => eval infix.1 + eval infix.2
                     to Sub => eval infix.1 - eval infix.2
